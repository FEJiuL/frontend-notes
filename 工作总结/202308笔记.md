```javascript
/**
 * == 顺序执行promise ==
 * @param {*} funcs - 执行队列
 * @returns
 */
export function runPromiseInSequence(funcs) {
  return function (ctx = {}) {
    function next(data) {
      let fn = funcs.shift();
      if (fn) {
        try {
          let ret = fn(ctx);
          // promise
          if (ret && ret.then) return ret.then(next);
          // sync function
          else return next(ret);
        } catch (err) {
          return Promise.reject(err);
        }
      }
      // 执行栈为空则返回
      else return Promise.resolve(data);
    }
    return next();
  };
}
```
